    In src/store/config.ts, add to the store:
    ```typescript
    const markedDays = ref<Set<string>>(new Set())

    function toggleVacationDay(dateStr: string) {
      const next = new Set(markedDays.value)
      if (next.has(dateStr)) {
        next.delete(dateStr)
      } else {
        next.add(dateStr)
      }
      markedDays.value = next  // assignment required — Vue 3 does NOT track Set.add/delete mutations
    }
    ```
    Return markedDays and toggleVacationDay alongside existing exports.

    CRITICAL: Never mutate markedDays.value directly with .add()/.delete(). Always create a new Set and reassign to trigger Vue reactivity.
